### 2.2 改进

扩展 `isOwner()` 支持多种身份识别：

```typescript
// src/tools/policy/owner-check.ts
export function isOwner(externalUserId: string): boolean {
  const ownerIds = env.OWNER_USER_IDS.split(",").map(id => id.trim());
  return ownerIds.includes(externalUserId);
}
```

## 3. 限流（Rate Limit）

### 3.1 目标

防止公开入口被滥用，保护系统资源。

### 3.2 策略

```text
Owner：不限流
普通用户：每分钟 5 条，每天 100 条
匿名用户：每分钟 3 条，每天 50 条
```

### 3.3 实现

使用简单的内存限流（基于 Map）：

```typescript
// src/rate-limit/rate-limit.service.ts
export class RateLimitService {
  private limits = new Map<string, { count: number; resetAt: number }>();

  check(key: string, maxPerMinute: number): boolean {
    const now = Date.now();
    const record = this.limits.get(key);

    if (!record || now > record.resetAt) {
      this.limits.set(key, { count: 1, resetAt: now + 60000 });
      return true;
    }

    if (record.count >= maxPerMinute) {
      return false; // 超过限制
    }

    record.count++;
    return true;
  }

  cleanup() {
    const now = Date.now();
    for (const [key, record] of this.limits.entries()) {
      if (now > record.resetAt) {
        this.limits.delete(key);
      }
    }
  }
}
```

### 3.4 集成

在 `chat.service.ts` 中检查限流：

```typescript
export async function chat(input: ChatInput): Promise<ChatOutput> {
  // 检查限流
  if (!isOwner(input.user_id)) {
    const rateLimitKey = `chat:${input.user_id}`;
    const allowed = rateLimitService.check(rateLimitKey, 5);
    if (!allowed) {
      throw new Error("请求过于频繁，请稍后再试");
    }
  }

  // ... 原有逻辑
}
```

### 3.5 环境变量

```env
# Rate Limit

RATE_LIMIT_PER_MINUTE=5
RATE_LIMIT_PER_DAY=100
RATE_LIMIT_OWNER_BYPASS=true
```

## 4. 错误日志

### 4.1 目标

记录系统错误，方便排查问题。

### 4.2 实现

使用 Fastify 自带的 logger + 简单的文件日志：

```typescript
// src/app.ts
const app = Fastify({
  logger: {
    level: env.LOG_LEVEL || "info",
    file: env.LOG_FILE || "./logs/app.log",
  },
});

// 全局错误处理
app.setErrorHandler((error, request, reply) => {
  app.log.error({
    err: error,
    url: request.url,
    method: request.method,
    userId: request.userId,
  }, "Request error");

  reply.status(500).send({
    error: "Internal Server Error",
    message: error.message,
  });
});
```

### 4.3 环境变量

```env
# Logging
LOG_LEVEL="info"
LOG_FILE="./logs/app.log"
ERROR_LOG_ENABLED=true
```

## 5. 备份

### 5.1 目标

定期备份数据库和 persona 文件，防止数据丢失。

### 5.2 备份内容

- PostgreSQL 数据库
- `persona/` 目录
- `.env` 文件（可选）

### 5.3 备份脚本

```bash
#!/bin/bash
# scripts/backup.sh

BACKUP_DIR="./backups"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_PATH="$BACKUP_DIR/backup_$TIMESTAMP"

mkdir -p "$BACKUP_PATH"

# 备份数据库
pg_dump $DATABASE_URL > "$BACKUP_PATH/database.sql"

# 备份 persona
cp -r persona "$BACKUP_PATH/"

# 备份 .env（可选）
cp .env "$BACKUP_PATH/.env.backup"

# 压缩
tar -czf "$BACKUP_PATH.tar.gz" -C "$BACKUP_DIR" "backup_$TIMESTAMP"
rm -rf "$BACKUP_PATH"

echo "Backup created: $BACKUP_PATH.tar.gz"

# 清理旧备份（保留最近 7 天）
find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +7 -delete
```

### 5.4 定时备份

使用 cron 定时执行：

```bash
# 每天凌晨 3 点备份
0 3 * * * cd /path/to/lusiyuan-core && ./scripts/backup.sh
```

### 5.5 恢复脚本

```bash
#!/bin/bash
# scripts/restore.sh

BACKUP_FILE=$1

if [ -z "$BACKUP_FILE" ]; then
  echo "Usage: ./scripts/restore.sh <backup_file.tar.gz>"
  exit 1
fi

# 解压
tar -xzf "$BACKUP_FILE" -C ./backups/

BACKUP_DIR=$(basename "$BACKUP_FILE" .tar.gz)

# 恢复数据库
psql $DATABASE_URL < "./backups/$BACKUP_DIR/database.sql"

# 恢复 persona
cp -r "./backups/$BACKUP_DIR/persona" ./

echo "Restore completed from: $BACKUP_FILE"
```

### 5.6 环境变量

```env
# Backup
BACKUP_ENABLED=true
BACKUP_DIR="./backups"
BACKUP_RETENTION_DAYS=7
```

## 6. Health Check

### 6.1 目标

提供健康检查端点，方便监控系统状态。

### 6.2 基础健康检查

```typescript
// src/routes/health.route.ts
export async function healthRoutes(app: FastifyInstance) {
  app.get("/health", async (request, reply) => {
    return {
      ok: true,
      service: "lusiyuan-core",
      version: env.APP_VERSION || "1.0.0",
      timestamp: new Date().toISOString(),
    };
  });
}
```

### 6.3 深度健康检查

```typescript
app.get("/health/deep", async (request, reply) => {
  const checks: Record<string, string> = {};

  // 检查数据库
  try {
    await prisma.$queryRaw`SELECT 1`;
    checks.database = "ok";
  } catch (err) {
    checks.database = "error";
  }

  // 检查 pgvector
  try {
    await prisma.$queryRaw`SELECT 1 FROM "MemoryEmbedding" LIMIT 1`;
    checks.pgvector = "ok";
  } catch (err) {
    checks.pgvector = "error";
  }

  // 检查模型提供商
  try {
    await modelProvider.chat([
      { role: "user", content: "test" }
    ]);
    checks.model_provider = "ok";
  } catch (err) {
    checks.model_provider = "error";
  }

  const allOk = Object.values(checks).every(v => v === "ok");

  return {
    ok: allOk,
    checks,
    timestamp: new Date().toISOString(),
  };
});
```

## 11. 部署建议

### 11.1 生产环境检查清单

部署到生产环境前，确保：

- [ ] 配置了 `OWNER_USER_IDS`
- [ ] 启用了 `RATE_LIMIT_ENABLED=true`
- [ ] 配置了日志文件路径
- [ ] 设置了定时备份（cron）
- [ ] 测试了 `/health` 和 `/health/deep` 端点
- [ ] Web 前端显示了隐私提示
- [ ] 数据库连接使用了生产环境凭据
- [ ] `.env` 文件不在 git 仓库中

### 11.2 监控建议

- 使用 `curl http://your-domain/health` 定期检查服务状态
- 使用 `tail -f logs/app.log` 查看实时日志
- 定期检查 `backups/` 目录确保备份正常

### 11.3 故障恢复

如果系统出现问题：

1. 查看日志：`tail -100 logs/app.log`
2. 检查健康状态：`curl http://localhost:3000/health/deep`
3. 如果数据损坏，从备份恢复：`./scripts/restore.sh backups/backup_YYYYMMDD_HHMMSS.tar.gz`

## 总结

这是一个**轻量级的生产稳定化方案**，只做陆思源真正需要的功能：

- 保护 owner 权限
- 防止滥用（限流）
- 记录错误（排查问题）
- 备份数据（防止丢失）
- 监控状态（健康检查）
- 隐私提示（告知用户）

不做复杂的权限系统、审计日志、成本统计、评测框架等企业级功能。这些对个人项目来说太重了。
